Modest Maps JS

The following div is a map initialized like so:

    var provider = new MapProvider(function(coord) {
        var img = parseInt(coord.zoom) +'-r'+ parseInt(coord.row) +'-c'+ parseInt(coord.column) + '.png';
        return 'http://osm-bayarea.s3.amazonaws.com/' + img;
    });
    var map = new Map('map', provider, new Point(640,480))
    map.setCenterZoom(new Location(37.804656, -122.263606), 14);
    map.draw();
